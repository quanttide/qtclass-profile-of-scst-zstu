WITH daily stats AS(
    SELECT
        date,
        category_id,
        sum(price* sales)/ sum(sales) AS weighted_price
    FROM fact_price
    WHERE is_valid=1
    GROUP BY date, category_id
)
SELECT
    date,
    category_id,
    weighted price,
    100* weighted_price/
        lagInFrame(weighted_price) OVER(PARTITION BY category_id ORDER BY date) 
        AS daily_index
FROM daily stats

(2)综合指数计算:

WITH base_year_data AS(
    SELECT
        category_id,
        sum(price* sales)/ sum(sales) AS base_price
    FROM fact price
    WHERE toYear(date)=2020--基期年份
    GROUP BY category_id
),
current data AS(
    SELECT
        date,
        category_id,
        sum(price* sales)/ sum(sales) AS current price
    FROM fact price
    WHERE is valid=1
    GROUP BY date, category_id
)
SELECT
    date,
    100* sum(current_price* b.base_price)/
        sum(b.base price* b.base price) AS composite index
FROM current data c
JOIN base_year_data b ON c.category_id= b.category_id
GROUP BY date
ORDER BY date
