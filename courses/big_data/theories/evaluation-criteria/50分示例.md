# price_index_calc.py
from clickhouse_driver import Client
import pandas as pd
from config import *

# 初始化 ClickHouse客户端
client = Client(
    host=CLICKHOUSE_HOST,
    port=CLICKHOUSE_PORT,
    user=CLICKHOUSE_USER,
    password=CLICKHOUSE_PASSWORD,
    database=CLICKHOUSE_DATABASE
)

def get_price_data():
    # 拉取所有价格数据(实际项目中可以按时间分页)
    sql = """  
    SELECT item_id, category_id, price, quantity, date
    FROM price_daily
    """
    data = client.execute(sql)
    df = pd.DataFrame(data, columns=["item_id","category_id","price","quantity","date"])
    df["date"] = pd.to_datetime(df["date"])
    return df

def calculate_price_index(df: pd.DataFrame, base_date: str = None):
    if base_date is None:
        base_date = df["date"].min().strftime("%Y-%m-%d")
    
    base_df = df[df["date"] == base_date]
    base_df = base_df[["item_id","price","quantity"]].rename(columns={
        "price": "base_price",
        "quantity": "base_quantity"
    })
    
    # 合并基期信息
    merged = df.merge(base_df, on="item_id", how="inner")
    
    # 计算每日加权 CPI
    merged["numerator"] = merged["price"] * merged["base_quantity"]
    merged["denominator"] = merged["base_price"] * merged["base_quantity"]
    
    daily_index = merged.groupby("date").agg({
        "numerator": "sum",
        "denominator": "sum"
    }).reset_index()
    daily_index["cpi"] = (daily_index["numerator"] / daily_index["denominator"]) * 100
    daily_index = daily_index[["date","cpi"]]
    return daily_index

def save_price_index_to_clickhouse(result_df):
    # 写入 ClickHouse(需要提前建好 price_index表)
    for _, row in result_df.iterrows():
        client.execute(
            "INSERT INTO price_index(day, cpi) VALUES",
            [(row["date"].date(), float(row["cpi"]))]
    print("☑☑CPI写入完成")

def run_cpi_calculation():
    df = get_price_data()
    result_df = calculate_price_index(df)
    print(result_df)
    save_price_index_to_clickhouse(result_df)
