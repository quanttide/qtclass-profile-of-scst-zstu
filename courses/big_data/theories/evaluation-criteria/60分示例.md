# src/data_loader.py
import oss2
from configparser import ConfigParser

def upload_to_oss():
    config= ConfigParser()
    config.read('config/settings.ini')
    auth= oss2.Auth(
        config['OSS']['ACCESS_KEY_ID'],
        config['OSS']['ACCESS_KEY_SECRET']
    )
    bucket= oss2.Bucket(
        auth,
        f'http://{config['OSS']['ENDPOINT']}",
        config['OSS']['BUCKET_NAME']
    )
    #上传示例文件
    bucket.put_object_from_file('data/category.csv','data/category.csv')
    bucket.put_object_from_file('data/price.csv','data/price.csv')
    print("数据上传完成")

# src/ch_operations.py
from clickhouse_driver import Client
from configparser import ConfigParser

class ClickHouseClient:
    def_init_(self):
        config= ConfigParser()
        config.read('config/settings.ini')
        self.client= Client(
            host='your_clickhouse_host',#如 service.clickhouse.cloud
            user='default',
            password='your_password',
            port=9440,#安全端口
            secure=True# SSL加密
        )
    def execute_query(self, query):
        return self.client.execute(query)

### 四、价格指数核心实现

### 1.加权价格计算

# src/index_calculator.py
from datetime import datetime, timedelta
import pandas as pd

class PriceIndexCalculator:
    def_init_(self, ch_client):
        self.client= ch_client
    def calculate_daily_index(self, base_date='2023-01-01'):
        query=f"""
        SELECT
            date,
            category_id,
            sum(price* sales)/ sum(sales) AS weighted_price,
            100*(sum(price* sales)/ sum(sales))/
            (SELECT sum(price* sales)/ sum(sales)
            FROM price
            WHERE date='{base_date}' AND category_id= p.category_id
            ) AS price_index
        FROM price p
        GROUP BY date, category_id
        ORDER BY date
        """
        return self.client.execute_query(query)

### 2.可视化实现

# src/visualization.py
import matplotlib.pyplot as plt
import pandas as pd

def plot_index_trend(data, save_path='results/price_trend.png'):
    df= pd.DataFrame(data, columns=['date','category','price','index'])
    df['date']= pd.to_datetime(df['date'])
    plt.figure(figsize=(12, 6))
    for category in df['category'].unique():
        subset= df[df['category']== category]
        plt.plot(subset['date'], subset['index'], label=category)
    plt.title('Daily Price Index by Category')
    plt.xlabel('Date')
    plt.ylabel('Index(Base 100)')
    plt.legend()
    plt.grid()
    plt.savefig(save_path)
    plt.close()
